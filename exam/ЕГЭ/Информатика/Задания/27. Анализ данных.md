def dist(dot1: tuple, dot2: tuple) -> float:  
    return ((dot2[0] - dot1[0]) ** 2 + (dot2[1] - dot1[1]) ** 2) ** 0.5  
  
  
def dbscan(array: list) -> list:  
    def get_cluster(target: tuple):  
        result = []  
        for dot in array:  
            if dist(target, dot) < 1:  
                result.append(dot)  
                array.remove(dot)  
  
        for dot in result:  
            result.extend(get_cluster(dot))  
  
        return result  
  
    db = []  
    while array:  
        db.append(get_cluster(array[0]))  
    return db

    data = read_file('27A_25448.txt')  
    clusters = [cluster for cluster in dbscan(data) if len(cluster) > 1]
