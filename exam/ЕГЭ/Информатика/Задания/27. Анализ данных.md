```python
from math import dist  
  
  
def dbscan(array: list) -> list:  
    def get_cluster(target: tuple):  
        result = []  
        for dot in array:  
            if dist(dot, target) < 1:  
                result.append(dot)  
                array.remove(dot)  
  
        for dot in result:  
            if target != dot:  
                result.extend(get_cluster(dot))  
  
        return result  
  
    db = []  
    while array:  
        db.append(get_cluster(array[0]))  
  
    return db
```
